//User function template for C++
class Solution{
public:	
	int findKRotation(int arr[], int n) {
	   
	   int start = 0;
	   int end = n-1;
	   
	   while(start <= end){
	       
	       int mid = start + (end-start)/2;
	       int leftMid = (mid+n-1)%n;
	       int rightMid = (mid+1)%n;
	       if(arr[mid] <= arr[leftMid] && arr[mid] <= arr[rightMid]){
	           return mid;
	       }
	       
	       else if((arr[start] <= arr[mid]) && (arr[start] > arr[end])){
	           start = mid+1;
	       }
	       
	       else{
	            end = mid-1;    
	       }
	   }
	   return 0;
	}

};

int main() {
    int t;

        int n, i;
        cin >> n;
        int a[n];
        for (i = 0; i < n; i++) {
            cin >> a[i];
        }
        Solution ob;
        auto ans = ob.findKRotation(a, n);
        cout << ans << "\n";
    }
    return 0;
}
